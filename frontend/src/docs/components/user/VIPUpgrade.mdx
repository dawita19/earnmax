---
title: VIP Upgrade Component
description: Calculates upgrade costs and benefits between VIP levels
---

```tsx live
interface VIPLevel {
  level: number;
  investment: number;
  dailyEarning: number;
}

const VIPUpgrade = ({ currentLevel, levels }: { currentLevel: number; levels: VIPLevel[] }) => {
  const [selectedLevel, setSelectedLevel] = useState<number | null>(null);

  const calculateUpgrade = (targetLevel: number) => {
    if (targetLevel <= currentLevel) return null;
    
    const current = levels.find(l => l.level === currentLevel);
    const target = levels.find(l => l.level === targetLevel);
    
    if (!current || !target) return null;
    
    return {
      cost: target.investment - current.investment,
      dailyIncrease: target.dailyEarning - current.dailyEarning,
      roiDays: Math.ceil((target.investment - current.investment) / 
               (target.dailyEarning - current.dailyEarning))
    };
  };

  return (
    <div className="space-y-6">
      <h3 className="font-medium">VIP Upgrade Calculator</h3>
      
      <div className="grid grid-cols-3 md:grid-cols-9 gap-2">
        {levels.map(level => (
          <button
            key={level.level}
            onClick={() => setSelectedLevel(level.level)}
            disabled={level.level <= currentLevel}
            className={`p-2 rounded-md text-center ${
              level.level === currentLevel
                ? 'bg-gray-200 cursor-default'
                : level.level > currentLevel
                ? level.level === selectedLevel
                  ? 'bg-blue-100 border-2 border-blue-500'
                  : 'bg-blue-50 hover:bg-blue-100'
                : 'bg-gray-100 text-gray-400 cursor-default'
            }`}
          >
            <div className="font-bold">VIP {level.level}</div>
            <div className="text-xs">
              {level.investment.toLocaleString()} ETB
            </div>
          </button>
        ))}
      </div>

      {selectedLevel && selectedLevel > currentLevel && (
        <div className="p-4 bg-green-50 rounded-lg">
          <h4 className="font-medium mb-2">
            Upgrade from VIP {currentLevel} to VIP {selectedLevel}
          </h4>
          
          {(() => {
            const upgrade = calculateUpgrade(selectedLevel);
            return upgrade ? (
              <div className="space-y-2 text-sm">
                <div className="flex justify-between">
                  <span>Upgrade Cost:</span>
                  <span className="font-medium">
                    {upgrade.cost.toLocaleString()} ETB
                  </span>
                </div>
                <div className="flex justify-between">
                  <span>Daily Earnings Increase:</span>
                  <span className="font-medium">
                    +{upgrade.dailyIncrease.toLocaleString()} ETB
                  </span>
                </div>
                <div className="flex justify-between">
                  <span>ROI Period:</span>
                  <span className="font-medium">
                    {upgrade.roiDays} days
                  </span>
                </div>
                <button className="mt-3 w-full bg-green-600 text-white py-2 rounded-md hover:bg-green-700">
                  Proceed with Upgrade
                </button>
              </div>
            ) : null;
          })()}
        </div>
      )}
    </div>
  );
};

// Example usage
<VIPUpgrade
  currentLevel={2}
  levels={[
    { level: 0, investment: 0, dailyEarning: 20 },
    { level: 1, investment: 1200, dailyEarning: 40 },
    { level: 2, investment: 3000, dailyEarning: 100 },
    { level: 3, investment: 6000, dailyEarning: 200 },
    { level: 4, investment: 12000, dailyEarning: 400 }
  ]}
/>
