            vipMaxWithdrawal={vipDetails.maxTotalWithdrawal}
          />

          <div className="rounded-lg border bg-card p-6">
            <h3 className="text-lg font-semibold mb-4">Earnings Overview</h3>
            <EarningsChart data={earningsData} />

        </div>
