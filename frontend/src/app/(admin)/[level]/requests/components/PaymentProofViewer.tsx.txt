import { useState } from 'react';
import Image from 'next/image';
import { Dialog, DialogContent, DialogTrigger } from '@/components/ui/dialog';
import { OCRResultDisplay } from './OCRResultDisplay';

interface PaymentProofViewerProps {
  proofUrl: string;
  ocrData?: {
    amount: number;
    accountNumber: string;
    date: string;
    confidence: number;
  };
}

export function PaymentProofViewer({ proofUrl, ocrData }: PaymentProofViewerProps) {
  const [isOpen, setIsOpen] = useState(false);

  return (
    <Dialog open={isOpen} onOpenChange={setIsOpen}>
      <DialogTrigger asChild>
        <Button variant="outline" size="sm">
          View Proof
        </Button>
      </DialogTrigger>
      <DialogContent className="max-w-4xl">
        <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
          <div className="relative h-96 bg-gray-100 rounded-md overflow-hidden">
            <Image
              src={proofUrl}
              alt="Payment proof"
              fill
              className="object-contain"
              quality={100}
              priority
            />
          </div>
          <div className="space-y-4">
            <h3 className="font-semibold">Payment Verification</h3>
            {ocrData ? (
              <OCRResultDisplay data={ocrData} />
            ) : (
              <div className="text-sm text-muted-foreground">
                OCR processing not available
              </div>
            )}
            <div className="space-y-2">
              <div className="flex items-center gap-2">
                <Checkbox id="amount-verified" />
                <Label htmlFor="amount-verified">Amount Verified</Label>
              </div>
              <div className="flex items-center gap-2">
                <Checkbox id="account-verified" />
                <Label htmlFor="account-verified">Account Verified</Label>
              </div>
              <div className="flex items-center gap-2">
                <Checkbox id="date-verified" />
                <Label htmlFor="date-verified">Date Verified</Label>
              </div>
            </div>
          </div>
        </div>
      </DialogContent>
    </Dialog>
  );
}

function OCRResultDisplay({ data }: { data: PaymentProofViewerProps['ocrData'] }) {
  return (
    <div className="space-y-2 text-sm">
      <div className="flex justify-between">
        <span className="font-medium">Amount:</span>
        <span>
          {data?.amount.toLocaleString()} ETB
          <span className="text-muted-foreground ml-2">
            ({data?.confidence}% confidence)
          </span>
        </span>
      </div>
      <div className="flex justify-between">
        <span className="font-medium">Account:</span>
        <span>{data?.accountNumber}</span>
      </div>
      <div className="flex justify-between">
        <span className="font-medium">Date:</span>
        <span>{data?.date}</span>
      </div>
    </div>
  );
}
