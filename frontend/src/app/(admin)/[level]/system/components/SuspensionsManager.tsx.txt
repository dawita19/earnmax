import { getActiveSuspensions, suspendUser, unsuspendUser } from '@/lib/api/admin';
import { Badge } from '@/components/ui/badge';

export async function SuspensionOverview() {
  const suspensions = await getActiveSuspensions();

  return (
    <div className="bg-white rounded-lg shadow p-4">
      <div className="flex justify-between items-center mb-4">
        <h3 className="text-lg font-medium">Active Suspensions</h3>
        <span className="inline-flex items-center px-3 py-1 rounded-full text-sm font-medium bg-red-100 text-red-800">
          {suspensions.length} Active
        </span>
      </div>

      <div className="space-y-4">
        {suspensions.length === 0 ? (
          <p className="text-gray-500 text-center py-4">No active suspensions</p>
        ) : (
          suspensions.map((suspension) => (
            <div key={suspension.suspension_id} className="border rounded-lg p-4 hover:bg-gray-50">
              <div className="flex justify-between items-start">
                <div>
                  <h4 className="font-medium">User #{suspension.user_id}</h4>
                  <p className="text-sm text-gray-500 mt-1">{suspension.reason}</p>
                  <div className="mt-2 flex space-x-2">
                    <Badge variant={suspension.status === 'active' ? 'destructive' : 'secondary'}>
                      {suspension.status}
                    </Badge>
                    <span className="text-xs text-gray-500">
                      {new Date(suspension.start_date).toLocaleDateString()}
                    </span>
                  </div>
                </div>
                <div className="flex space-x-2">
                  <form action={unsuspendUser.bind(null, suspension.suspension_id)}>
                    <button
                      type="submit"
                      className="px-3 py-1 text-sm bg-green-600 text-white rounded hover:bg-green-700"
                    >
                      Lift
                    </button>
                  </form>
                  <form action={suspendUser.bind(null, suspension.user_id)}>
                    <input type="hidden" name="reason" value="Extended suspension" />
                    <button
                      type="submit"
                      className="px-3 py-1 text-sm bg-yellow-600 text-white rounded hover:bg-yellow-700"
                    >
                      Extend
                    </button>
                  </form>
                </div>
              </div>
            </div>
          ))
        )}
      </div>

      <div className="mt-4">
        <button className="w-full py-2 border border-dashed border-gray-300 rounded-md hover:bg-gray-50 text-gray-500">
          + Add New Suspension
        </button>
      </div>
    </div>
  );
}
